\begin{tcolorbox}[
  enhanced,
  colback=blue!3!white,
  colframe=blue!55!black,
  fonttitle=\bfseries,
  title={Key Idea — Order of Operations (PEMDAS)},
  rounded corners,
  drop shadow,
  breakable
]
\textbf{Always do operations in this order:}
\begin{enumerate}
  \item \textbf{P}arentheses (grouping symbols)
  \item \textbf{E}xponents
  \item \textbf{M}ultiplication and \textbf{D}ivision \emph{from left to right}
  \item \textbf{A}ddition and \textbf{S}ubtraction \emph{from left to right}
\end{enumerate}
\vspace{0.25em}
\textbf{Important:} Multiplication and division are a \emph{pair} (same priority). Addition and subtraction are a \emph{pair} (same priority).
\end{tcolorbox}

\begin{tcolorbox}[
  enhanced,
  colback=orange!6!white,
  colframe=orange!70!black,
  fonttitle=\bfseries,
  title={Common Mistakes to Avoid},
  rounded corners,
  drop shadow,
  breakable
]
\begin{itemize}
  \item Doing addition before multiplication. Example: $3 + 2\cdot 5$ is $3+10=13$, not $(3+2)\cdot 5$.
  \item Forgetting left-to-right for $\div$ and $\times$. Example: $24\div 3\cdot 2$ is $(24\div 3)\cdot 2=16$.
  \item Losing negative signs. Rewrite subtraction as addition: $a-b=a+(-b)$.
  \item Forgetting parentheses after a minus sign: $-(x-7)= -x+7$.
\end{itemize}
\end{tcolorbox}

\subsection*{Worked Examples (follow the order)}

\textbf{Example 1.} Evaluate: $6+3\cdot 4$
\[
6+3\cdot 4 = 6+12 = 18.
\]

\textbf{Example 2.} Evaluate: $24\div 3\cdot 2$
\[
24\div 3\cdot 2 = (24\div 3)\cdot 2 = 8\cdot 2 = 16.
\]

\textbf{Example 3.} Evaluate: $-(5-12)+3^2$
\[
-(5-12)+3^2 = -(-7)+9 = 7+9=16.
\]

\textbf{Example 4.} Simplify: $2\bigl(3x-(x-4)\bigr)$
\[
2(3x-(x-4)) = 2(3x-x+4)=2(2x+4)=4x+8.
\]
