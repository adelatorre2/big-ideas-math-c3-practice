\begin{itemize}
  \item \textbf{Integers \& signs:} adding/subtracting negatives; multiplying/dividing signs; rewriting subtraction.
  \item \textbf{Distributive property:} $a(b+c)=ab+ac$ and $-(b-c)=-b+c$.

% ------------------------------
% Part A — Evaluate & Simplify (Warm-Up)
% ------------------------------
\subsection*{Part A — Evaluate \& Simplify (Warm-Up)}
Evaluate or simplify.

\begin{enumerate}
  \item $8 - 3\cdot(5-7)$ \vspace{0.9cm}
  \item $-12 + 4\cdot 3 - 6$ \vspace{0.9cm}
  \item $24\div 3\cdot 2 - 5$ \vspace{0.9cm}
  \item $-(6-14) + 2^3$ \vspace{0.9cm}
  \item $-3\bigl(2 - (5-9)\bigr)$ \vspace{1.0cm}
  \item $5 - \bigl(3 - (2-7)\bigr)$ \vspace{1.0cm}
  \item $-2^3 + 10$ \vspace{0.9cm}
  \item $(-2)^3 + 10$ \vspace{0.9cm}
\end{enumerate}

% ------------------------------
% Part B — Distribute, Then Combine Like Terms
% ------------------------------
\subsection*{Part B — Distribute, Then Combine Like Terms}
Simplify completely.

\begin{enumerate}
  \item $-(x+7) + 3x$ \vspace{1.0cm}
  \item $2(3x-5) - (x-4)$ \vspace{1.1cm}
  \item $-(4x-9) - 2(\,x+3\,)$ \vspace{1.1cm}
  \item $5 - 3\bigl(2x-1\bigr)$ \vspace{1.1cm}
  \item $-2\bigl(3y-8\bigr) + (y-5)$ \vspace{1.2cm}
  \item $4\bigl(2a-(a-6)\bigr)$ \vspace{1.2cm}
  \item $-\bigl(2m-3n+4\bigr) + 2(m+n)$ \vspace{1.3cm}
\end{enumerate}

\newpage

% ------------------------------
% Part C — Solve Equations (One Variable)
% ------------------------------
\subsection*{Part C — Solve Equations (Watch the Signs)}
Solve for the variable. Check by substitution when you can.

\begin{enumerate}
  \item $x - 9 = -4$ \vspace{1.0cm}
  \item $-7 = x + 3$ \vspace{1.0cm}
  \item $-3x = 24$ \vspace{1.0cm}
  \item $\dfrac{x}{-6} = -5$ \vspace{1.0cm}
  \item $2x - 5 = 11$ \vspace{1.1cm}
  \item $-4x + 9 = -19$ \vspace{1.1cm}
  \item $3(x-4)= -18$ \vspace{1.2cm}
  \item $-2(\,x+7\,) = 10$ \vspace{1.2cm}
  \item $5 - (x-3) = -4$ \vspace{1.2cm}
  \item $2\bigl(3x-1\bigr) - 4 = 8$ \vspace{1.3cm}
\end{enumerate}

\newpage

% ------------------------------
% Part D — Multi-Step Simplify & Solve
% ------------------------------
\subsection*{Part D — Multi-Step (All Skills Together)}
Simplify first, then solve.

\begin{enumerate}
  \item $3 - \bigl(2x-7\bigr) = 10$ \vspace{1.3cm}
  \item $-\bigl(4x-5\bigr) + 2x = 13$ \vspace{1.3cm}
  \item $2\bigl(x-6\bigr) - 3\bigl(x+1\bigr) = -8$ \vspace{1.4cm}
  \item $5\bigl(2x-3\bigr) + 4 = 3\bigl(3x+2\bigr)$ \vspace{1.5cm}
  \item $\dfrac{1}{2}(\,6x-8\,) - (x-3) = 5$ \vspace{1.6cm}
\end{enumerate}

\subsection*{Challenge (optional)}
These are harder. Take your time and write neat steps.

\begin{enumerate}
  \item A student earns $\$12$ per hour. They worked 5 hours, but had a $\$8$ fee taken out. Write and evaluate an expression for their pay.
  \vspace{1.7cm}

  \item The temperature was $-3^\circ$F. Overnight it dropped 7 degrees, then rose 5 degrees. What is the final temperature?
  \vspace{1.4cm}

  \item Solve and interpret: $-2(\,x-4\,) + 3 = 17$. (What number makes the equation true?)
  \vspace{1.6cm}

  \item True or false? Explain: $-(a-b)= -a-b$.
  \vspace{1.7cm}
\end{enumerate}
