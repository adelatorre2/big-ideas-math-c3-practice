\subsection{0.1 Integers \& Signs}
\label{sol:0.1-integers-and-signs}

\subsubsection*{Practice A — Integer Operations}
\begin{enumerate}[label=\arabic*.]
  \item $7 + (-3) = 4$
  \item $-8 + (-6) = -14$
  \item $-15 + 9 = -6$
  \item $12 - 19 = -7$
  \item $-4 - 11 = -15$
  \item $-13 - (-5) = -13 + 5 = -8$
  \item $18 - (-7) = 18 + 7 = 25$
  \item $(-6)(-5) = 30$
  \item $(9)(-4) = -36$
  \item $\dfrac{-42}{6} = -7$
  \item $\dfrac{-36}{-9} = 4$
  \item $\lvert -17 \rvert = 17$
\end{enumerate}

\subsubsection*{Practice B — Decode the Signs}
\begin{enumerate}[label=\arabic*.]
  \item $10 - (-3) + (-2) = 10 + 3 - 2 = 11$
  \item $-5 - (-8) - 4 = -5 + 8 - 4 = -1$
  \item $-12 + 6 - (-9) = -12 + 6 + 9 = 3$
  \item $3 - 7 + (-11) = 3 - 7 - 11 = -15$
  \item $-20 - (-6) + 2 = -20 + 6 + 2 = -12$
  \item $-1 - (-1) - (-1) = -1 + 1 + 1 = 1$
\end{enumerate}

\subsubsection*{Practice C — Distribute with Negatives}
\begin{enumerate}[label=\arabic*.]
  \item $-(x + 9) = -x - 9$
  \item $-(4x - 7) = -4x + 7$
  \item $-3(2x - 5) = -6x + 15$
  \item $2 - (x - 6) = 2 - x + 6 = 8 - x$
  \item $-(2x + 3y) = -2x - 3y$
  \item $-(a - b + c) = -a + b - c$
  \item $-5(3m + 2) - (-4m) = (-15m - 10) + 4m = -11m - 10$
\end{enumerate}

\subsubsection*{Practice D — Simplify Expressions}
\begin{enumerate}[label=\arabic*.]
  \item $-7x + 3x = -4x$
  \item $8y - (3y - 5) = 8y - 3y + 5 = 5y + 5$
  \item $-(2x - 3) + (x - 10) = (-2x + 3) + x - 10 = -x - 7$
  \item $4a - 2(a - 6) = 4a - 2a + 12 = 2a + 12$
  \item $-3(2p - 1) + 5p = (-6p + 3) + 5p = -p + 3$
  \item $6 - 2(4 - x) = 6 - (8 - 2x) = 2x - 2$
\end{enumerate}

\subsubsection*{Challenge (optional)}
\begin{enumerate}[label=\arabic*.]
  \item $-(2x - 5) - (3x + 4) + 9 = (-2x + 5) + (-3x - 4) + 9 = -5x + 10$
  \item $-(4 - (2y - 7)) = -(4 - 2y + 7) = -(11 - 2y) = 2y - 11$
  \item $3\bigl(-(x - 2)\bigr) - 2(x - 6) = 3(-x + 2) - (2x - 12) = -5x + 18$
  \item A number that is $-8$ units from $0$ can be $8$ or $-8$.
\end{enumerate}

\subsection{0.2 Order of Operations}
\label{sol:0.2-order-of-operations}

\subsubsection*{Practice A — Evaluate (Warm-Up)}
\begin{enumerate}[label=\arabic*.]
  \item $8 + 2 \cdot 6 = 8 + 12 = 20$
  \item $30 - 5 \cdot 4 = 30 - 20 = 10$
  \item $18 \div 3 + 7 = 6 + 7 = 13$
  \item $4^2 - 3 \cdot 5 = 16 - 15 = 1$
  \item $12 \div 2 \cdot 3 = (12 \div 2)\cdot 3 = 6\cdot 3 = 18$
  \item $36 \div (6 \cdot 3) = 36 \div 18 = 2$
  \item $7(5 - 2) + 1 = 7\cdot 3 + 1 = 22$
  \item $9 - (2 + 4)^2 = 9 - 6^2 = 9 - 36 = -27$
\end{enumerate}

\subsubsection*{Practice B — Parentheses \& Negatives}
\begin{enumerate}[label=\arabic*.]
  \item $-(8 - 13) + 6 = -(-5) + 6 = 11$
  \item $-3(2 - 7) = -3(-5) = 15$
  \item $5 - (3 - 9) = 5 - (-6) = 11$
  \item $2\bigl(4 - (1 - 6)\bigr) = 2\bigl(4 - (-5)\bigr) = 2\cdot 9 = 18$
  \item $-2^3 + 10 = -(2^3) + 10 = -8 + 10 = 2$
  \item $(-2)^3 + 10 = -8 + 10 = 2$
\end{enumerate}

\subsubsection*{Practice C — Simplify (Variables)}
\begin{enumerate}[label=\arabic*.]
  \item $3x + 2x\cdot 4 = 3x + 8x = 11x$
  \item $10 - 2(3x + 1) = 10 - (6x + 2) = 8 - 6x$
  \item $4(x + 2) - 3(x - 5) = (4x + 8) - (3x - 15) = x + 23$
  \item $2\bigl(3x - (x - 7)\bigr) = 2(3x - x + 7) = 2(2x + 7) = 4x + 14$
  \item $6 - 2(4 - x) = 6 - (8 - 2x) = 2x - 2$
  \item $-(2x - 3) + 5(x - 1) = (-2x + 3) + (5x - 5) = 3x - 2$
\end{enumerate}

\subsubsection*{Challenge (optional)}
\begin{enumerate}[label=\arabic*.]
  \item $48 \div 6 \cdot 2 + 3^2 = (48 \div 6)\cdot 2 + 9 = 8\cdot 2 + 9 = 25$
  \item $2\bigl(5 - (3x - 2)\bigr) - (x - 4) = 2(7 - 3x) - x + 4 = 18 - 7x$
  \item Because division and multiplication have the same priority, you work left-to-right: $24 \div 3 \cdot 2 = (24 \div 3)\cdot 2$.
\end{enumerate}

\subsubsection*{Practice B — Distribute with Negatives and Parentheses}
\begin{enumerate}[label=\arabic*.]

  \item $-3(4x + 2) = -12x - 6$
  \item $2 - (x - 5) = 2 - x + 5 = 7 - x$

\subsubsection*{Practice C — Combine Like Terms After Distributing}
\begin{enumerate}[label=\arabic*.]
  \item $2(x + 5) + x = (2x + 10) + x = 3x + 10$
  \item $3(x - 4) + 2x = (3x - 12) + 2x = 5x - 12$
  \item $4(x + 1) - 2(x - 3) = (4x + 4) - (2x - 6) = 2x + 10$
  \item $-3(x - 2) + 5(x + 1) = (-3x + 6) + (5x + 5) = 2x + 11$
  \item $2(2x - 1) + 3(x + 4) = (4x - 2) + (3x + 12) = 7x + 10$
  \item $5(x - 3) - 2(3x - 7) = (5x - 15) - (6x - 14) = -x - 1$

\subsubsection*{Practice D — Factor (Distribute Backwards)}

  \item $14x + 21 = 7(2x + 3)$
  \item $6x - 42 = 6(x - 7)$

\subsubsection*{Practice E — Spot the Distributive Property}

  \item $10(0.3x + 0.7) = 3x + 7$

\subsection{0.4 Mixed Foundations Review}
\label{sol:0.4-mixed-foundations-review}

\subsubsection*{Part A — Evaluate \& Simplify (Warm-Up)}
\begin{enumerate}[label=\arabic*.]
  \item $8 - 3\cdot(5 - 7) = 8 - 3\cdot(-2) = 8 + 6 = 14$
  \item $-12 + 4\cdot 3 - 6 = -12 + 12 - 6 = -6$
  \item $24 \div 3 \cdot 2 - 5 = (24 \div 3)\cdot 2 - 5 = 8\cdot 2 - 5 = 11$
  \item $-(6 - 14) + 23 = -(-8) + 23 = 31$
  \item $-3\bigl(2 - (5 - 9)\bigr) = -3\bigl(2 - (-4)\bigr) = -3\cdot 6 = -18$
  \item $5 - \bigl(3 - (2 - 7)\bigr) = 5 - \bigl(3 - (-5)\bigr) = 5 - 8 = -3$
  \item $-2^3 + 10 = -(2^3) + 10 = -8 + 10 = 2$
  \item $(-2)^3 + 10 = -8 + 10 = 2$
\end{enumerate}

\subsubsection*{Part B — Distribute, Then Combine Like Terms}
\begin{enumerate}[label=\arabic*.]
  \item $-(x + 7) + 3x = -x - 7 + 3x = 2x - 7$
  \item $2(3x - 5) - (x - 4) = (6x - 10) - x + 4 = 5x - 6$
  \item $-(4x - 9) - 2(x + 3) = (-4x + 9) + (-2x - 6) = -6x + 3$
  \item $5 - 3(2x - 1) = 5 - (6x - 3) = 8 - 6x$
  \item $-2(3y - 8) + (y - 5) = (-6y + 16) + y - 5 = -5y + 11$
  \item $4\bigl(2a - (a - 6)\bigr) = 4(a + 6) = 4a + 24$
  \item $-(2m - 3n + 4) + 2(m + n) = (-2m + 3n - 4) + (2m + 2n) = 5n - 4$
\end{enumerate}

\subsubsection*{Part C — Solve Equations (Watch the Signs)}
\begin{enumerate}[label=\arabic*.]
  \item $x - 9 = -4 \Rightarrow x = 5$
  \item $-7 = x + 3 \Rightarrow x = -10$
  \item $-3x = 24 \Rightarrow x = -8$
  \item $\dfrac{x}{-6} = -5 \Rightarrow x = 30$
  \item $2x - 5 = 11 \Rightarrow 2x = 16 \Rightarrow x = 8$
  \item $-4x + 9 = -19 \Rightarrow -4x = -28 \Rightarrow x = 7$
  \item $3(x - 4) = -18 \Rightarrow x - 4 = -6 \Rightarrow x = -2$
  \item $-2(x + 7) = 10 \Rightarrow x + 7 = -5 \Rightarrow x = -12$
  \item $5 - (x - 3) = -4 \Rightarrow 5 - x + 3 = -4 \Rightarrow 8 - x = -4 \Rightarrow x = 12$
  \item $2(3x - 1) - 4 = 8 \Rightarrow 6x - 6 = 8 \Rightarrow 6x = 14 \Rightarrow x = \dfrac{7}{3}$
\end{enumerate}

\subsubsection*{Part D — Multi-Step (All Skills Together)}
\begin{enumerate}[label=\arabic*.]
  \item $3 - (2x - 7) = 10 \Rightarrow 3 - 2x + 7 = 10 \Rightarrow 10 - 2x = 10 \Rightarrow x = 0$
  \item $-(4x - 5) + 2x = 13 \Rightarrow -4x + 5 + 2x = 13 \Rightarrow -2x = 8 \Rightarrow x = -4$
  \item $2(x - 6) - 3(x + 1) = -8 \Rightarrow (2x - 12) - (3x + 3) = -8 \Rightarrow -x - 15 = -8 \Rightarrow x = -7$
  \item $5(2x - 3) + 4 = 3(3x + 2) \Rightarrow 10x - 11 = 9x + 6 \Rightarrow x = 17$
  \item $\dfrac12(6x - 8) - (x - 3) = 5 \Rightarrow (3x - 4) - x + 3 = 5 \Rightarrow 2x - 1 = 5 \Rightarrow x = 3$
\end{enumerate}
