\textbf{Distribute (multiply) to every term inside the parentheses.}

\vspace{0.4em}
\textbf{Form:}\quad $a(b+c)=ab+ac$ \quad and \quad $a(b-c)=ab-ac$.

\vspace{0.4em}
\textbf{Helpful memory:} \emph{Whatever is outside the parentheses gets multiplied by \emph{each} term inside.}
\end{tcolorbox}

\begin{itemize}
  \item \textbf{Forgetting a term:} $3(x+5)$ must become $3x+15$ (not just $3x+5$).
  \item \textbf{Sign mistakes:} $-2(x-4)= -2x+8$.
  \item \textbf{Parentheses matter:} $-(x-7)= -x+7$.
  \item \textbf{Combining unlike terms too early:} Distribute first, then simplify.
\end{itemize}
\end{tcolorbox}

\subsection*{Worked Examples (Follow the Steps)}

\textbf{Example 1.} Expand: $4(\,x+3\,)$
\[
4(x+3)=4\cdot x+4\cdot 3=4x+12.
\]

\textbf{Example 2.} Expand: $-3(\,2x-5\,)$
\[
-3(2x-5)=(-3)\cdot 2x+(-3)\cdot(-5)=-6x+15.
\]

\textbf{Example 3.} Expand and simplify: $2(\,x+4\,)+3(\,x-1\,)$
\[
2(x+4)+3(x-1)=(2x+8)+(3x-3)=5x+5.
\]

\textbf{Example 4.} Factor using the distributive property (reverse direction): $6x+18$
\[
6x+18=6(x+3).
\]
