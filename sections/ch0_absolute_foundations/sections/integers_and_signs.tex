\begin{itemize}
  \item \textbf{Integers} are whole numbers and their opposites: $\dots,-3,-2,-1,0,1,2,3,\dots$
  \item \textbf{Opposites:} $a$ and $-a$ are opposites. Example: the opposite of $7$ is $-7$.
  \item \textbf{Absolute value:} $|a|$ is the distance from $0$. Always nonnegative.
  \item \textbf{Subtracting:} $a-b = a+(-b)$ (subtracting is adding the opposite).
  \item \textbf{Sign rules:}
    \begin{itemize}
      \item $(-)(-) = (+)$ and $(+)(-) = (-)$ and $(-)(+) = (-)$
      \item Same signs \(\Rightarrow\) add magnitudes and keep the sign.
      \item Different signs \(\Rightarrow\) subtract magnitudes and keep the sign of the larger magnitude.
    \end{itemize}
  \item \textbf{Distributing negatives:} 
    \[ -(a+b)= -a-b \qquad \text{and} \qquad -(a-b)= -a+b. \]
\end{itemize}
\end{tcolorbox}

\textbf{Example 1.} Rewrite subtraction as addition: 
\[ 6 - (-4) = 6 + 4 = 10. \]

\textbf{Example 2.} Different signs (subtract magnitudes):
\[ -12 + 5 = -(12-5) = -7. \]

\textbf{Example 3.} Distribute a negative:
\[ -(3x-8) = -3x + 8. \]

\textbf{Example 4.} Combine like terms carefully:
\[ -2x + 7x - (x-5) = 5x - x + 5 = 4x + 5. \]
\end{tcolorbox}
